\section{Áp dụng Computational Thinking}
\defaultauthor

\begin{itemize}
  \item \textbf{Decomposition:} chia hệ thống thành frontend, routers, agents, services, models và persistence.
  \item \textbf{Pattern Recognition:} nhận diện cấu trúc trip/day/leg, mẫu nearest-neighbor, mẫu context và hành vi đổi mode.
  \item \textbf{Abstraction:} đóng gói route, context và preference thành model/service có contract rõ ràng.
  \item \textbf{Algorithm Design:} xây dựng greedy scheduling, weighted scoring, adaptation và fallback.
\end{itemize}
